def make_index(tmp_path_factory: Any) -> Any:
    fn = tmp_path_factory.mktemp("pages") / "index.md"
    fn.write_text(
        textwrap.dedent(
            """
            ---
            templateKey: blog-post
            tags: ['python',]
            title:  My Awesome Post
            date: 2022-01-21T16:40:34
            status: draft

            ---

            This is my awesome post.
            """
        )
    )
    return tmp_path_factory
